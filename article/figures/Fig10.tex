\documentclass[border=10pt]{standalone}

  LabelStyle/.style = { rectangle, rounded corners, draw,
                       font = \bfseries },
  EdgeStyle/.append style = {->} }
\thispagestyle{empty}
\begin{document}
\begin{tikzpicture}
  \SetGraphUnit{5}
  \Vertex[x=0,y=0]{l2}
  \Vertex[x=0,y=2]{l1}

  \Vertex[x=8,y=0]{s2}
  \Vertex[x=8,y=2]{s1}
  
  \Vertex[x=2,y=1]{A}
  \Vertex[x=6,y=1]{B}
  
  \Edge[label = 1](l1)(A)
  \Edge[label = 10](l2)(A)

  
  \Edge[label = 10](B)(s1)
  \Edge[label = 1](B)(s2)
  
  \Edge(A)(B)
