  \Vertex[x=0,y=0]{s3}
  \Vertex[x=0,y=2]{s2}

  \Vertex[x=8,y=0]{t3}
  \Vertex[x=8,y=2]{t2}
  \Vertex[x=8,y=4]{t1}
  
  \Vertex[x=2,y=2]{cs}
  \Vertex[x=6,y=2]{ct}
  
  \Edge(s1)(cs)
  \Edge(s2)(cs)
  \Edge(s3)(cs)
  
  \Edge(ct)(t1)
  \Edge(ct)(t2)
  \Edge(ct)(t3)
  
  \Edge(cs)(ct)
