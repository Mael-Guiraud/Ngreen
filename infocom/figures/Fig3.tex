\documentclass[border=10pt]{standalone}
\usepackage{tkz-graph}

\tikzset{
  LabelStyle/.style = { rectangle, rounded corners, draw,
                       font = \bfseries },
  VertexStyle/.append style = {font = \Large\bfseries},
  EdgeStyle/.append style = {<-} }
\thispagestyle{empty}
\begin{document}
\begin{tikzpicture}
  \SetGraphUnit{5}
  
  \Vertex[x=4,y=1]{s}
  \Vertex[x=0,y=1]{l}
  \SetVertexNoLabel
  \Vertex[x=2,y=2]{A}
  \Vertex[x=2,y=-0]{B}
  \Edge[label = 3](l)(B)
  \Edge[label = 4](B)(s)

  \Edge[label = 3](l)(A)
  \Edge[label = 2](A)(s)

  \tikzset{EdgeStyle/.append style = {bend left = 50}}
\end{tikzpicture}

\end{document}
